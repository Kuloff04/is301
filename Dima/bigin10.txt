# Begin10: Даны два ненулевых числа. Найти сумму, разность, произведение и частное их квадратов.

a = float(input("Введите первое ненулевое число a: "))
b = float(input("Введите второе ненулевое число b: "))

a_sq = a ** 2 # Квадрат a
b_sq = b ** 2 # Квадрат b

s_sum = a_sq + b_sq
s_diff = a_sq - b_sq
s_prod = a_sq * b_sq
s_quot = a_sq / b_sq

print(f"Сумма квадратов: {s_sum}")
print(f"Разность квадратов: {s_diff}")
print(f"Произведение квадратов: {s_prod}")
print(f"Частное квадратов: {s_quot}")
