# Begin9: Даны два неотрицательных числа a и b. Найти их среднее геометрическое: sqrt(a * b).
import math # Импортируем модуль math для функции sqrt

a = float(input("Введите первое неотрицательное число a: "))
b = float(input("Введите второе неотрицательное число b: "))

# Проверка на неотрицательность
if a < 0 or b < 0:
    print("Ошибка: Числа должны быть неотрицательными!")
else:
    # Вариант 1: Использование функции math.sqrt()
    average_geometric = math.sqrt(a * b)
    
    # Вариант 2: Использование возведения в степень 0.5
    # average_geometric = (a * b) ** 0.5 

    print(f"Среднее геометрическое: {average_geometric}")
