# Получение длин ребер от пользователя
try:
    a = float(input("Введите длину ребра a: "))
    b = float(input("Введите длину ребра b: "))
    c = float(input("Введите длину ребра c: "))
    
    # Проверка на положительные значения
    if a <= 0 or b <= 0 or c <= 0:
        print("Длины ребер должны быть положительными числами.")
    else:
        # 1. Вычисление объема (V = a * b * c)
        V = a * b * c
        
        # 2. Вычисление площади поверхности (S = 2 * (a*b + b*c + a*c))
        # Площадь поверхности равна сумме площадей 6 граней (по две одинаковых)
        S = 2 * (a * b + b * c + a * c)
        
        # Вывод результатов
        print("\n--- Результаты ---")
        print(f"Длины ребер: a={a}, b={b}, c={c}")
        print(f"Объем параллелепипеда (V): {V}")
        print(f"Площадь поверхности параллелепипеда (S): {S}")

except ValueError:
    print("Некорректный ввод. Пожалуйста, введите числовые значения для длин ребер.")
