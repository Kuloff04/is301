import math

try:
    a, b = map(float, input("Введите два неотрицательных числа a и b через пробел: ").split())
    
    if a < 0 or b < 0:
        print("Ошибка: Оба числа должны быть неотрицательными.")
    else:
        """ Вычисление среднего геометрического: корень из a * b """
        geometric_mean = math.sqrt(a * b)
        
        print(f"Среднее геометрическое (√a·b): {geometric_mean}")

except ValueError:
    print("Некорректный ввод. Пожалуйста, введите два числовых значения.")
